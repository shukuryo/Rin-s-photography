            font-family: 'Inter', sans-serif;

        }

        .slide {

            transition: opacity 1s ease-in-out;

        }

        /* アクティブなドットのスタイル */

        .dot.active {

            background-color: white;

        }

<body class="bg-gray-900 text-white">

 

    <!-- メインコンテンツ: スライドショー -->

    <main class="relative h-screen overflow-hidden">

        <!-- スライドのコンテナ -->

        <div id="slideshow-container" class="w-full h-full">

            <!-- ここにJavaScriptでスライドが挿入されます -->

        </div>

       

        <!-- 画像を少し暗くするためのオーバーレイ -->

        <div class="absolute inset-0 bg-black bg-opacity-20 z-10 pointer-events-none"></div>

 

        <!-- スライドショーのナビゲーションボタン -->

        <button id="prev" class="absolute left-4 top-1/2 -translate-y-1/2 z-20 p-2 bg-black bg-opacity-30 rounded-full hover:bg-opacity-50 transition-all">

            <svg class="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M15 19l-7-7 7-7"></path></svg>

        </button>

        <button id="next" class="absolute right-4 top-1/2 -translate-y-1/2 z-20 p-2 bg-black bg-opacity-30 rounded-full hover:bg-opacity-50 transition-all">

            <svg class="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M9 5l7 7-7 7"></path></svg>

        </button>

 

        <!-- ドットインジケーター -->

        <div id="dots-container" class="absolute bottom-8 left-1/2 -translate-x-1/2 z-20 flex space-x-2">

             <!-- ここにJavaScriptでドットが挿入されます -->

        </div>

    </main>

 

    <script>

        document.addEventListener('DOMContentLoaded', function() {

            // --- スライドショーの画像設定 ---

            // Placeholder.coの画像を使用しています。お好きな画像URLに変更してください。

            const images = [

                'https://placehold.co/1920x1080/000000/FFFFFF?text=Slide+1',

                'https://placehold.co/1920x1080/1c1c1c/FFFFFF?text=Slide+2',

                'https://placehold.co/1920x1080/383838/FFFFFF?text=Slide+3',

                'https://placehold.co/1920x1080/545454/FFFFFF?text=Slide+4',

                'https://placehold.co/1920x1080/7a7a7a/FFFFFF?text=Slide+5',

                'https://placehold.co/1920x1080/969696/FFFFFF?text=Slide+6',

                'https://placehold.co/1920x1080/b8b8b8/000000?text=Slide+7',

                'https://placehold.co/1920x1080/dcdcdc/000000?text=Slide+8'

            ];

 

            const slideshowContainer = document.getElementById('slideshow-container');

            const dotsContainer = document.getElementById('dots-container');

 

            // --- 画像とドットを動的に生成 ---

            images.forEach((src, index) => {

                // スライド要素を作成

                const slide = document.createElement('div');

                slide.className = 'slide absolute w-full h-full opacity-0';

                slide.style.backgroundImage = `url('${src}')`;

                slide.style.backgroundSize = 'cover';

                slide.style.backgroundPosition = 'center';

                if (index === 0) {

                    slide.classList.remove('opacity-0');

                    slide.classList.add('opacity-100');

                }

                slideshowContainer.appendChild(slide);

 

                // ドット要素を作成

                const dot = document.createElement('button');

                dot.className = 'dot w-3 h-3 bg-gray-500 rounded-full hover:bg-white transition-colors';

                if (index === 0) {

                    dot.classList.add('active');

                }

                dot.addEventListener('click', () => {

                    showSlide(index);

                    resetInterval();

                });

                dotsContainer.appendChild(dot);

            });

           

            // --- スライドショーのロジック ---

            const slides = slideshowContainer.querySelectorAll('.slide');

            const dots = dotsContainer.querySelectorAll('.dot');

            const prevBtn = document.getElementById('prev');

            const nextBtn = document.getElementById('next');

 

            let currentIndex = 0;

            let slideInterval;

 

            function showSlide(index) {

                // インデックスの境界チェック

                if (index >= slides.length) {

                    currentIndex = 0;

                } else if (index < 0) {

                    currentIndex = slides.length - 1;

                } else {

                    currentIndex = index;

                }

 

                // すべてのスライドを非表示にし、ドットを非アクティブにする

                slides.forEach(slide => {

                    slide.classList.remove('opacity-100');

                    slide.classList.add('opacity-0');

                });

                dots.forEach(dot => {

                    dot.classList.remove('active');

                });

 

                // 現在のスライドとドットをアクティブにする

                slides[currentIndex].classList.remove('opacity-0');

                slides[currentIndex].classList.add('opacity-100');

                dots[currentIndex].classList.add('active');

            }

 

            function nextSlide() {

                showSlide(currentIndex + 1);

            }

 

            function prevSlide() {

                showSlide(currentIndex - 1);

            }

 

            // --- イベントリスナー ---

            prevBtn.addEventListener('click', () => {

                prevSlide();

                resetInterval();

            });

            nextBtn.addEventListener('click', () => {

                nextSlide();

                resetInterval();

            });

           

            // --- 自動再生 ---

            function startInterval() {

                slideInterval = setInterval(nextSlide, 5000); // 5秒ごとにスライドを切り替え

            }

 

            function resetInterval() {

                clearInterval(slideInterval);

                startInterval();

            }

 

            startInterval();

        });

    </script>

</body>
